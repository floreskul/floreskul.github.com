[

],

300
],
[
"Brazil",
400
],
[
"Canada",
500
],
[
"France",
600
],
[
"RU",
700
]
]);

var options = {
"region": "150",
"displayMode": "regions"
};
// -- end of generated code

var chart = new google.visualization.GeoChart(document.getElementById('chart_div'));
chart.draw(data, options);
};
